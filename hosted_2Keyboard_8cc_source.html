119 void HostedKeyboard::blocking(bool enable)
120 {
121  int flags = fcntl(0, F_GETFL);
122  if (flags < 0)
123  {
124  return;
125  }
126  if (enable)
127  flags &= ~O_NONBLOCK;
128  else
129  flags |= O_NONBLOCK;
130  fcntl(0, F_SETFL, flags);
131 }
