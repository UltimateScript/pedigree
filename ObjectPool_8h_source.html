104  // We only add the object back to the pool if we aren't already at
105  // capacity (otherwise we'd resize the Vector).
106  m_Pool.reserve(poolSize, true);
107  if (m_Pool.count() < poolSize)
108  {
109  m_Pool.pushBack(object);
110  }
