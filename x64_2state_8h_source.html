465 uintptr_t X64SyscallState::getSyscallParameter(size_t n) const
466 {
467  switch (n)
468  {
469  case 0:
470  return m_Rbx;
471  case 1:
472  return m_Rdx;
473  case 2:
474  return m_Rsi;
475  case 3:
476  return m_Rdi;
477  case 4:
478  return m_R8;
479  case 5:
480  return m_R9;
481 
482  // redirection for linux syscall parameters
483  case 6:
484  return m_Rdi;
485  case 7:
486  return m_Rsi;
487  case 8:
488  return m_Rdx;
489  case 9:
490  return m_R10;
491  case 10:
492  return m_R8;
493  case 11:
494  return m_R9;
495 
496  default:
497  WARNING("Bad syscall parameter requested: " << Dec << n);
498  return 0;
499  }
