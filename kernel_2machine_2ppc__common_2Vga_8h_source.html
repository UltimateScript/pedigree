37 // or 32 bpp specific form.

41  (((r & 0x1f) << 11) | ((g & 0x1f) << 6) | ((b & 0x1f) << 1))
42 #define RGB_32(r, g, b) (0 | r << 24 | g << 16 | b << 8)
43 #else
