65 bool MaltaSerial::isConnected()
66 {
67  return true;
68  uint8_t nStatus = m_pRegs->mstat;
69  // Bits 0x30 = Clear to send & Data set ready.
70  // Mstat seems to be 0xFF when the device isn't present.
71  if ((nStatus & 0x30) && nStatus != 0xFF)
72  return true;
73  else
74  return false;
75 }
