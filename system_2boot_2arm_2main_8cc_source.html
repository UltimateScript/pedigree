92 void writeHex(unsigned int n)
93 {
94  bool noZeroes = true;
95 
96  int i;
97  unsigned int tmp;
98  for (i = 28; i > 0; i -= 4)
99  {
100  tmp = (n >> i) & 0xF;
101  if (tmp == 0 && noZeroes)
102  {
103  continue;
104  }
105 
106  if (tmp >= 0xA)
107  {
108  noZeroes = false;
109  writeChar(tmp - 0xA + 'a');
110  }
111  else
112  {
113  noZeroes = false;
114  writeChar(tmp + '0');
115  }
116  }
117 
118  tmp = n & 0xF;
119  if (tmp >= 0xA)
120  {
121  writeChar(tmp - 0xA + 'a');
122  }
123  else
124  {
125  writeChar(tmp + '0');
126  }
127 }

206 {

208  // handlers

218  Elf32 elf("kernel");

229  bs.size = elf.m_pHeader->shentsize;
