39 static int preloadThread(void *p)
40 {
41  const char *s = reinterpret_cast<const char *>(p);
42 
43  NOTICE("PRELOAD: " << s);
44 
45  File *pFile = VFS::instance().find(String(s));
46  if (pFile)
47  {
48  NOTICE("PRELOAD: preloading " << s << "...");
49  size_t sz = 0;
50  while (sz < pFile->getSize())
51  {
52  pFile->read(sz, 0x1000, 0);
53  sz += 0x1000;
54  }
55  }
56 
57  NOTICE("PRELOAD: preload " << s << " has completed.");
58  g_Preloads.release();
59 
60  return 0;
61 }
